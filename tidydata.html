import pandas as pd

def showProfile(df) :
    col_profile = pd.concat([df.dtypes, df.isna().sum()/len(df)*100, df.sample().T], axis=1 ).reset_index()
    col_profile.columns = ['列名', '类型', '缺失比%','样例'] 
    print(df.shape)  
    return (col_profile)

    cols = df.columns
    for i in range(2, len(cols)):
        sel = df.iloc[:, [0,1,i]]
        fname = f'./mydata/yjbg_y2022_{cols[i]}.pkl.gz'
        sel.to_pickle(fname)
        print(sel.columns)
